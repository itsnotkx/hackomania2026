Check if reconnect with exponential backoff was implemented in Phase 2. If it was not implemented or is incomplete, add it now. The full pattern:

```js
const RECONNECT_DELAYS = [1000, 2000, 4000, 8000];
let reconnectAttempt = 0;
let keepaliveTimer = null;

function connect() {
  ws = new WebSocket(BACKEND_URL);
  ws.binaryType = 'arraybuffer';

  ws.onopen = () => {
    reconnectAttempt = 0;
    setBadgeState('ANALYZING');
    startKeepalive();
  };

  ws.onclose = () => {
    clearInterval(keepaliveTimer);
    setBadgeState('DISCONNECTED');
    const delay = RECONNECT_DELAYS[Math.min(reconnectAttempt, RECONNECT_DELAYS.length - 1)];
    reconnectAttempt++;
    setTimeout(connect, delay);
  };
}

function startKeepalive() {
  clearInterval(keepaliveTimer);
  keepaliveTimer = setInterval(() => {
    if (ws && ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify({ type: 'ping' }));
    }
  }, 15000);
}
```

Add E2E latency tracking to overlay.js:
```js
let chunkId = 0;
const chunkTimestamps = new Map();

// When sending a chunk (in the audio capture callback):
const id = chunkId++;
chunkTimestamps.set(id, Date.now());
ws.send(JSON.stringify({ type: 'chunk_start', chunk_id: id }));
ws.send(audioBuffer); // existing binary send

// In ws.onmessage:
const result = JSON.parse(event.data);
if (result.chunk_id !== undefined && chunkTimestamps.has(result.chunk_id)) {
  const latency = Date.now() - chunkTimestamps.get(result.chunk_id);
  console.log(`[E2E] chunk ${result.chunk_id}: ${latency}ms`);
  chunkTimestamps.delete(result.chunk_id);
}
```

NOTE: The backend must echo `chunk_id` back in its response for this to work. If the backend does not currently echo it, add the echo: in `backend/main.py` (or equivalent), read the preceding `chunk_start` JSON frame, store `chunk_id`, and include it in the verdict response JSON as `{ label, score, ms, chunk_id }`. If the backend architecture makes this difficult, fall back to a simpler approach: timestamp the send, timestamp the receive on the next message, and log the delta (without per-chunk correlation).

```ts
const RECONNECT_DELAYS = [1000, 2000, 4000, 8000];
const reconnectAttempt = useRef(0);
const keepaliveTimer = useRef<NodeJS.Timeout | null>(null);
const wsRef = useRef<WebSocket | null>(null);

function connect() {
  const ws = new WebSocket(BACKEND_URL);
  ws.binaryType = 'arraybuffer';
  wsRef.current = ws;

  ws.onopen = () => {
    reconnectAttempt.current = 0;
    setBadgeDisplayState('ANALYZING');
    keepaliveTimer.current = setInterval(() => {
      if (wsRef.current?.readyState === WebSocket.OPEN) {
        wsRef.current.send(JSON.stringify({ type: 'ping' }));
      }
    }, 15000);
  };

  ws.onclose = () => {
    if (keepaliveTimer.current) clearInterval(keepaliveTimer.current);
    setBadgeDisplayState('DISCONNECTED');
    const delay = RECONNECT_DELAYS[Math.min(reconnectAttempt.current, RECONNECT_DELAYS.length - 1)];
    reconnectAttempt.current++;
    setTimeout(connect, delay);
  };
}
```
