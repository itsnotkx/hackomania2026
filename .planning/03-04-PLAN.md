```tsx
import React, { useEffect, useRef, useState } from 'react';
import { AppState, AppStateStatus, PermissionStatus, Platform, View } from 'react-native';
import { check, request, PERMISSIONS, RESULTS } from 'react-native-permissions';
import FloatButton from 'react-native-float-bubble';
import { BadgeState } from './src/components/overlay/BadgeOverlay';
import PermissionSetupScreen from './src/components/permissions/PermissionSetupScreen';
import { startAudioCapture, stopAudioCapture } from './src/services/AudioCaptureService';
import { useWebSocket } from './src/hooks/useWebSocket';
import VIForegroundService from '@supersami/rn-foreground-service';

const BACKEND_WS_URL = 'ws://YOUR_RAILWAY_URL/ws'; // Replace with actual deployed URL from Phase 1 summary

type PermissionsState = 'checking' | 'missing' | 'granted';

export default function App() {
  const [permissionsState, setPermissionsState] = useState<PermissionsState>('checking');
  const [isMonitoring, setIsMonitoring] = useState(false);
  const { badgeState, connect, disconnect, sendChunk, connectionState } = useWebSocket(BACKEND_WS_URL);

  // Check both permissions on mount
  useEffect(() => {
    checkPermissions();
  }, []);

  // Update float bubble content whenever badge state changes
  useEffect(() => {
    if (permissionsState === 'granted') {
      const label = badgeState === 'AI' ? 'AI DETECTED' : badgeState === 'HUMAN' ? 'HUMAN' : badgeState;
      FloatButton.updateFloatButton?.({ text: label }).catch(() => {});
    }
  }, [badgeState, permissionsState]);

  const checkPermissions = async () => {
    const micStatus = await check(PERMISSIONS.ANDROID.RECORD_AUDIO);
    const overlayStatus = await check(PERMISSIONS.ANDROID.SYSTEM_ALERT_WINDOW);
    if (micStatus === RESULTS.GRANTED && overlayStatus === RESULTS.GRANTED) {
      setPermissionsState('granted');
      showFloatBubble();
    } else {
      setPermissionsState('missing');
    }
  };

  const showFloatBubble = () => {
    FloatButton.showFloatButton();
    FloatButton.onPress(() => {
      // Float button tap: bring app to foreground to show expanded card (Phase 4)
    });
  };

  const handlePermissionsGranted = async () => {
    setPermissionsState('granted');
    showFloatBubble();
  };

  const startMonitoring = async () => {
    setIsMonitoring(true);
    // Start foreground service notification (keeps service alive in background)
    await VIForegroundService.startService({
      id: 1001,
      title: 'NoScam',
      text: 'Monitoring for AI-generated speech...',
      icon: 'ic_launcher',
      importance: 'low',
    });
    connect();
    // Wire audio capture output directly to WebSocket sender
    startAudioCapture(sendChunk);
  };

  const stopMonitoring = async () => {
    setIsMonitoring(false);
    stopAudioCapture();
    disconnect();
    await VIForegroundService.stopService();
  };

  if (permissionsState === 'checking') {
    return <View style={{ flex: 1, backgroundColor: '#111' }} />;
  }

  if (permissionsState === 'missing') {
    return (
      <PermissionSetupScreen onPermissionsGranted={handlePermissionsGranted} />
    );
  }

  // Main monitoring screen — shown briefly on first launch, then user goes to another app
  // The float bubble badge is the primary UI while monitoring
  return (
    <View style={{ flex: 1, backgroundColor: '#111', justifyContent: 'center', alignItems: 'center' }}>
      {/* This view is only visible when the user returns to the app.
          The float bubble is the primary always-on-top UI. */}
    </View>
  );
}
```

Key wiring points:
1. `startAudioCapture(sendChunk)` — this is the critical link: audio chunks from the microphone flow directly into `sendChunk` (from `useWebSocket`) as binary WebSocket frames
2. `FloatButton.showFloatButton()` — activates the TYPE_APPLICATION_OVERLAY floating bubble after permissions granted
3. `FloatButton.updateFloatButton()` — called in `useEffect` whenever `badgeState` changes to keep the float bubble label current
4. `VIForegroundService.startService()` — starts the notification that keeps the audio foreground service alive when the app is backgrounded
5. `PermissionSetupScreen` — gating the monitoring flow behind permission checks

Replace `'ws://YOUR_RAILWAY_URL/ws'` with the actual Railway URL recorded in the Phase 1 summary (01-03-SUMMARY.md).

Note: If `FloatButton.updateFloatButton` is not available in the installed react-native-float-bubble version, use `FloatButton.setOption({ text: label })` or the equivalent method from the library's documentation. Check the library's API at `NoScamAndroid/node_modules/react-native-float-bubble/` before writing the update call.
